## 📺 画面表示用コード

```typescript
/** Example 1: Helper */
function safeAt<T>(items: T[], index: number): T | undefined {
  return items.at(index);
}

/** Example 2: guard */
function firstOrThrow<T>(items: T[]): T {
  const value = items[0];
  if (value === undefined) throw new Error("empty array");
  return value;
}

/** Example 3: fallback */
const result = safeAt(["ok"], 1) ?? "default";
```
