## 📺 画面表示用コード

```typescript
/** Example 1: 型ガード */
function ensureBigint(value: number | bigint): bigint {
  return typeof value === "bigint" ? value : BigInt(value);
}

/** Example 2: 単位管理 */
const MILLI = 1_000n;
const MICRO = 1_000_000n;

/** Example 3: JSON用ラッパー */
function serializeBigint(value: bigint) {
  return { value: value.toString() };
}
```
