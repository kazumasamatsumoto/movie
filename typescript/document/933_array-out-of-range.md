## 📺 画面表示用コード

```typescript
/** Example 1: undefined */
const list = ["a", "b"];
const missing = list[5]; // undefined

/** Example 2: guard */
const value = list[5];
if (value !== undefined) {
  console.log(value.toUpperCase());
}

/** Example 3: lengthチェック */
function getLast<T>(items: T[]): T | undefined {
  return items.length > 0 ? items[items.length - 1] : undefined;
}
```
