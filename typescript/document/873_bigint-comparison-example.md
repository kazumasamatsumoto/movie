## 📺 画面表示用コード

```typescript
/** Example 1: 閾値チェック */
function isLargeId(id: bigint): boolean {
  const limit = 10_000_000_000_000n;
  return id >= limit;
}

/** Example 2: 最新順ソート */
const ids = [5n, 2n, 100n];
ids.sort((a, b) => (a > b ? -1 : a < b ? 1 : 0));

/** Example 3: 範囲フィルタ */
const filtered = ids.filter((id) => id >= 10n && id <= 90n);
```
