---

## 📺 画面表示用コード

```typescript
/** Example 1: lengthチェック */
function firstSafe<T>(items: T[]): T | undefined {
  return items.length ? items[0] : undefined;
}

/** Example 2: at */
const first = [1, 2, 3].at(0);

/** Example 3: ユーティリティ */
function getOr<T>(items: T[], index: number, fallback: T): T {
  return items[index] ?? fallback;
}
```
