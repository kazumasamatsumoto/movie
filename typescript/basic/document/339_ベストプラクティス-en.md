/** Example 1: Type alias */
type Nullable<T> = T | null;
function findUser(id: number): Nullable<User> {
  return users.find(u => u.id === id) ?? null;
}

/** Example 2: Type guard */
if (user !== null) {
  console.log(user.name); // Safe
}

/** Example 3: Optional Chaining */
