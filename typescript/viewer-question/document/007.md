## 画面表示用コード

```typescript
enum RoleEnum {
  Admin = "admin",
  Editor = "editor",
}

const RoleLiteral = ["admin", "editor"] as const;
type Role = (typeof RoleLiteral)[number];

function label(role: Role) {
  return role === "admin" ? "管理者" : "編集";
}
```
