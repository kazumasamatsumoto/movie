## 画面表示用コード

```typescript
// デフォルト型パラメータの基本
interface ApiResponse<T = unknown> {
  data: T;
  status: number;
}

// 型引数を省略すると unknown になる
const response1: ApiResponse = { data: {}, status: 200 };
// 型引数を指定すると具体的な型になる
const response2: ApiResponse<User> = { data: user, status: 200 };
```

```typescript
// 実用例: ステート管理
class Store<T = Record<string, any>> {
  constructor(private state: T) {}

  getState(): T {
    return this.state;
  }
}

// デフォルト型で使用
const store1 = new Store({ count: 0 }); // Record<string, any>
// 明示的な型で使用
const store2 = new Store<AppState>({ count: 0, user: null });
```

```typescript
// 複数のデフォルト型パラメータ
type Result<T = unknown, E = Error> =
  | { success: true; value: T }
  | { success: false; error: E };

// 両方省略
const result1: Result = { success: true, value: 'hello' };
// 一部だけ指定
const result2: Result<string> = { success: false, error: new Error() };
```
