## 画面表示用コード

```typescript
type Result<T, E> = { ok: true; value: T } | { ok: false; error: E };

function success<T>(value: T): Result<T, never> {
  return { ok: true, value };
}

function failure<E>(error: E): Result<never, E> {
  return { ok: false, error };
}
```
