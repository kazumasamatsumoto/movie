## 画面表示用コード

```typescript
// tsconfig.json
{
  "compilerOptions": {
    "incremental": true,
    "tsBuildInfoFile": ".tsbuildinfo"
  }
}

// package.json
{
  "scripts": {
    "dev": "ts-node --esm src/index.ts",
    "build": "tsc -p tsconfig.json",
    "test": "ts-jest --passWithNoTests"
  }
}

import "ts-node/register";

console.log("dev server ready");
// nodemon --exec ts-node src/index.ts
```
