## 画面表示用コード

```typescript
type IsoDate = string;

function toIso(date: Date): IsoDate {
  return date.toISOString();
}

const nowIso = toIso(new Date());

function fromIso(value: IsoDate): Date {
  return new Date(value);
}

type Timeline = {
  createdAt: IsoDate;
  updatedAt: IsoDate;
};

const timeline: Timeline = { createdAt: nowIso, updatedAt: nowIso };
```
