## 画面表示用コード

```typescript
const roles = ["admin", "editor"] as const;

function accept(role: (typeof roles)[number]) {
  return role.toUpperCase();
}

accept("admin");

const config = {
  mode: "dark",
  retries: 3,
} as const;

type Mode = typeof config.mode;

function pickFlag<T extends readonly string[]>(flags: T) {
  return flags[0];
}

pickFlag(roles);
```
