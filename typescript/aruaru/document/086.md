## 画面表示用コード

```typescript
type FeatureFlag = {
  mode: "strict" | "relaxed";
  retries: number;
};

const flag = {
  mode: "strict",
  retries: 3,
  debug: true,
} satisfies FeatureFlag;

function enable(flag: FeatureFlag) {
  return flag.mode.toUpperCase();
}

enable(flag);
```
