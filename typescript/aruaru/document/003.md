## 画面表示用コード

```typescript
type Status = "draft" | "review" | "done";

function assertNever(x: never): never {
  throw new Error(`unexpected status: ${x}`);
}

function handleStatus(status: Status) {
  switch (status) {
    case "draft":
      return "WIP";
    case "review":
      return "Reviewing";
    case "done":
      return "Released";
    default:
      assertNever(status);
  }
}

handleStatus("draft");
handleStatus("done");
```
