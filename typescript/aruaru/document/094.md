## 画面表示用コード

```typescript
const roleLabel: Record<"admin" | "editor", string> = {
  admin: "管理者",
  editor: "編集者",
};

const flags: Record<string, boolean> = { feature: true };
flags.beta = false;

const typedFlags = {
  feature: true,
  beta: false,

```
