## 画面表示用コード

```typescript
interface User {
  id: string;
  name: string;
}

type UserDto = Pick<User, "id" | "name">;

type Handler = (user: User) => void;

const handler: Handler = user => {
  console.log(user.id);
};

interface Service {
  fetch(id: string): Promise<User>;
}

const service: Service = {
  async fetch(id) {
    return { id, name: "m" };
  },
};
```
