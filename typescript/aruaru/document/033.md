## 画面表示用コード

```typescript
type User = {
  id: string;
  role?: "admin" | "editor";
};

const candidate = { id: "u1", role: "admin" };

const strictUser = {
  id: "u1",
  role: "editor",
} satisfies User;

strictUser.role.toUpperCase();

function ensureUser(user: User) {
  if (!user.role) {
    throw new Error("role required");
  }
  return user;
}
```
