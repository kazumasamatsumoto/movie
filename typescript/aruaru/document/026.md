## 画面表示用コード

```typescript
type Event =
  | { type: "click"; x: number; y: number }
  | { type: "submit"; formId: string };

function handle(event: Event) {
  if (event.type === "click") {
    console.log(event.x, event.y);
  }
}

function exhaust(event: Event) {
  switch (event.type) {
    case "click":
      return "clicked";
    case "submit":
      return "submitted";
    default:
      const neverEvent: never = event;
      return neverEvent;
  }
}
```
