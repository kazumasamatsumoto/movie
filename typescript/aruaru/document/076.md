## 画面表示用コード

```typescript
import { z } from "zod";

const userSchema = z.object({
  id: z.string(),
  role: z.enum(["admin", "editor"]),
});

type User = z.infer<typeof userSchema>;

const parsed = userSchema.parse({ id: "u1", role: "admin" });

function handle(user: User) {
  return user.role.toUpperCase();
}
```
