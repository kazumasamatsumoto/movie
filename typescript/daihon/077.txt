## 📺 画面表示用コード

```typescript
// 配列のループ
let names: string[] = ["Alice", "Bob", "Charlie"];

// for...of文
for (let name of names) {
  console.log(name);
}

// 実用的な例
let userRoles: string[] = ["admin", "user", "guest"];
for (let role of userRoles) {
  console.log(`Role: ${role}`);
}

let productTags: string[] = ["本", "ペン", "ノート"];
for (let tag of productTags) {
  console.log(`Tag: ${tag}`);
}
```
