## 📺 画面表示用コード

```typescript
// Nest.jsのDTOとnumber型
import { IsNumber, IsOptional, Min, Max } from 'class-validator';
import { Transform } from 'class-transformer';

export class CreateProductDto {
  @IsNumber()
  @Min(0)
  price: number;

  @IsNumber()
  @Min(1)
  quantity: number;

  @IsOptional()
  @IsNumber()
  @Min(0)
  @Max(100)
  discount?: number;
}

// 実用的な例
export class UpdateUserDto {
  @IsNumber()
  @Min(0)
  @Max(120)
  age: number;
}
```
