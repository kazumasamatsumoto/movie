# #018 「CheckフェーズとsetImmediate」

```javascript
// immediate-vs-timeout.mjs
import fs from 'node:fs';
setTimeout(() => console.log('timer 0ms'), 0);
setImmediate(() => console.log('immediate'));
fs.readFile(new URL(import.meta.url), () => {
  setTimeout(() => console.log('timer inside I/O'), 0);
  setImmediate(() => console.log('immediate inside I/O'));
});
```

```javascript
// after-poll.mjs
import { setImmediate as afterPoll } from 'node:timers';
const ticket = afterPoll(() => console.log('runs after poll phase'));
console.log(`immediate handle: ${ticket}`);
```
