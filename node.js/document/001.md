# #001 「プロセスとスレッドの違いとは」

```javascript
// process-info.js
console.log(`PID: ${process.pid}`);
console.log(`PPID: ${process.ppid}`);
console.log(`Title: ${process.title}`);
```

```javascript
// cluster-worker.js
import cluster from 'node:cluster';
if (cluster.isPrimary) cluster.fork();
if (cluster.isWorker) console.log(`Worker ${process.pid} ready`);
```

```javascript
// worker-thread.js
import { Worker, isMainThread } from 'node:worker_threads';
if (isMainThread) new Worker(new URL(import.meta.url));
else console.log(`Thread ${process.threadId} running`);
```
