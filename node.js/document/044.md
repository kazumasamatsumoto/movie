```bash
sysctl -a | grep cache.size
```

```javascript
// share-cache.mjs
import cluster from 'node:cluster';
import os from 'node:os';
if (cluster.isPrimary) {
  for (let i = 0; i < 2; i += 1) cluster.fork({ WORKER: i });
} else {
  console.log(`worker ${process.env.WORKER} using shared L3 cache`);
}
```
