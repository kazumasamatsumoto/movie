# #020 「非ブロッキングI/Oの仕組み」

```javascript
// async-fetch.mjs
import https from 'node:https';
https.get('https://example.com', res => {
  res.on('data', () => {});
  res.on('end', () => console.log('response received without blocking event loop'));
});
console.log('other work continues');
```

```javascript
// await-demo.mjs
import { readFile } from 'node:fs/promises';
const files = [process.execPath, process.execPath];
const buffers = await Promise.all(files.map(file => readFile(file)));
console.log(buffers.map(buf => buf.byteLength));
```

```bash
NODE_DEBUG=http,fs node async-fetch.mjs
```
