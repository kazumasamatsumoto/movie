# #006 「スタックオーバーフローが起きる理由」

```javascript
// recursive-overflow.js
function recurse() {
  return recurse();
}
recurse(); // RangeError: Maximum call stack size exceeded
```

```javascript
// iterative-version.js
function sum(n) {
  let total = 0;
  for (let i = 1; i <= n; i++) total += i;
  return total;
}
console.log(sum(1_000_000));
```

```javascript
// chunked-task.js
let i = 0;
function step() {
  if (i >= 1e6) return console.log('done');
  do { i++; } while (i % 1000 !== 0);
  setImmediate(step);
}
step();
```
